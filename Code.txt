<script>
let inp = document.getElementById("input");
let trig = document.getElementById("trigger");
let findword = document.getElementById("find");
let replword = document.getElementById("repl");
trig.addEventListener("click",(e)=>{
if (replword.value == "\\n") {
  inp.value = inp.value.replaceAll(findword.value,"\n");
} else {
  inp.value = inp.value.replaceAll(findword.value,replword.value);
}
return;
});
</script>
